//Zmienne wyjsciowe
out vec2 iTexCoord;

void main(void) {
	vec4 tmp_vertex = vec4(vertex, 1.0);
	vec4 tmp_normal = vec4(normal, 1.0);

	gl_Position = P*V*M*tmp_vertex;

	iTexCoord = texCoord;
}
